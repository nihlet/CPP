int square(int num) {
    num = num * num;
    return num;
}

int square2(int &num) {
    num = num * num;
    return num;
}

float square(float num) {
    num = num * num;
    return num;
}

float square(float num, int add) {
    num = num * num + add;
    return num;
}

{
    num = num * num;

}

template <int a> int add(int num)
{
    num = num + a;
    return num;
}
